## Code

```cpp
#include <bits/stdc++.h>
using namespace std;

const int N = 1'000'005;
bool vi[N];
vector<int> in[N], ou[N];

void dfs(int u, vector<int>& p) {
    if (vi[u]) return;
    vi[u] = true;
    for (auto& v : ou[u]) dfs(v, p);
    p.push_back(u);
}

void dfs2(int u, vector<int>& co) {
    if (vi[u]) return;
    vi[u] = true;
    co.push_back(u); // place this above or below children dfs as you wish
    for (auto& v : in[u]) dfs2(v, co);
}

inline void solve() {
    int n, m; cin >> n >> m;
    for (int i = 0; i < m; ++i) {
        int u, v; cin >> u >> v;
        ou[u].push_back(v);
        in[v].push_back(u);
    }

    vector<int> p;
    for (int u = 1; u <= n; ++u) vi[u] = false;
    for (int u = 1; u <= n; ++u) dfs(u, p);
    reverse(p.begin(), p.end());
    
    for (int u = 1; u <= n; ++u) vi[u] = false;
    vector<vector<int>> scc;
    for (auto& u : p) {
        vector<int> co;
        dfs2(u, co);
        if (!co.empty()) scc.push_back(co);
    }
    
    for (auto& co : scc) {
        for (auto& u : co) {
            cout << u << ' ';
        }
        cout << endl;
    }
}
```
